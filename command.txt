command for list configuration
    >git config --list

command for adding user name
    >git coonfig --global user.name "username"

command for adding user email
    >git config --global user.email "userEmail@gamil.com"
    
command for initializing git
    >git init

command to list untracked files
    >git status

command for adding modified files to staging area
    >git add -A

command for adding all the staged files to local repository
    >git commit -m "commit message"


flages
    A => staged
    U => untraced
    M => modified

command for adding remote repository
    >git remote add name remote_path
    eg: remote add luminar https://github.com/reshroy/git-masterclass.git

command to check whether the remote repository is added
    >git remote -v

command for adding changes in loavl repository to remote repository
    >git push -u name branch_name
    eg: git push -u luminar master
